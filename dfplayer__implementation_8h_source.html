116static inline int dfplayer_get_volume(dfplayer_t *dev, uint8_t *volume)
117{
118 if (!volume) {
119 return -EINVAL;
120 }
121
122 uint16_t tmp;
123 int retval = dfplayer_query(dev, &tmp, DFPLAYER_CMD_GET_VOLUME);
124 if (retval) {
125 return retval;
126 }
127
128 *volume = (uint8_t)tmp;
129 return 0;
130}
