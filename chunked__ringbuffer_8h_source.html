    86static inline bool crb_start_chunk(chunk_ringbuf_t *rb)
    87{
    88 /* pointing to the start of the first chunk */
    89 if (rb->cur == rb->protect) {
    90 return false;
    91 }
    92
    93 rb->cur_start = rb->cur;
    94
    95 if (rb->protect == NULL) {

    99 return true;

    116static inline bool crb_add_byte(chunk_ringbuf_t *rb, uint8_t b)
    117{
    118 /* if this is the first chunk, protect will be at start */
    119 if (rb->cur == rb->protect &&
    120 rb->cur != rb->cur_start) {
    121 return false;
    122 }
    123
    124 *rb->cur = b;
    125
    126 /* handle wrap around */
    127 if (rb->cur == rb->buffer_end) {
    128 rb->cur = rb->buffer;
    129 } else {
    130 ++rb->cur;
    131 }
    132
    133 return true;
    134}
