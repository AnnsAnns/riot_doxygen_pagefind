    43int32_t _ihelp(int32_t x)
    44{
    45 int32_t y;
    46 static const int32_t qN = 13,
    47 qA = 12,
    48 B = 19900,
    49 C = 3516;
    50
    51 x = x << (31 - qN); /* Mask with PI */
    52 x = x >> (31 - qN); /* Note: SIGNED shift! (to qN) */
    53 x = x * x >> (2 * qN - 14); /* x=x^2 To Q14 */
    54
    55 y = B - (x * C >> 14); /* B - x^2*C */
    56 y = (1 << qA) /* A - x^2*(B-x^2*C) */
    57 - (x * y >> 16);
    58
    59 return y;
    60}

    102static inline unsigned sqrti(unsigned x)
    103{
    104 if (x <= 1) {
    105 return x;
    106 }
    107
    108 /* initial estimate */
    109 unsigned y0 = x >> 1;
    110 unsigned y1 = (y0 + x / y0) >> 1;
    111

    113 y0 = y1;
    114 y1 = (y0 + x / y0) >> 1;
