    328static inline clist_node_t *clist_remove(clist_node_t *list, clist_node_t *node)
    329{
    330 if (list->next) {
    331 if (list->next->next == node) {
    332 return clist_lpop(list);
    333 }
    334 else {
    335 clist_node_t *tmp = clist_find_before(list, node);
    336 if (tmp) {
    337 tmp->next = tmp->next->next;
    338 if (node == list->next) {
    339 list->next = tmp;
    340 }
    341 return node;
    342 }
    343 }
    344 }

    346 return NULL;
