    176uint8_t nrf24l01p_ng_etoval_aw(nrf24l01p_ng_aw_t address_width)
    177{
    178 if (address_width <= NRF24L01P_NG_AW_3BYTE) {
    179 return 3;
    180 }
    181 if (address_width == NRF24L01P_NG_AW_4BYTE) {
    182 return 4;
    183 }
    184 return 5;

    194static inline
    195nrf24l01p_ng_aw_t nrf24l01p_ng_valtoe_aw(uint8_t address_width)
    196{
    197 if (address_width <= 3) {
    198 return NRF24L01P_NG_AW_3BYTE;
    199 }
    200 if (address_width == 4) {
    201 return NRF24L01P_NG_AW_4BYTE;

    213static inline
    214uint16_t nrf24l01p_ng_etoval_ard(nrf24l01p_ng_ard_t retr_delay)
    215{

    229static inline
    230nrf24l01p_ng_ard_t nrf24l01p_ng_valtoe_ard(uint16_t retr_delay)
    231{

    233 return NRF24L01P_NG_ARD_4000US;

    235 if (retr_delay < 250) {
    236 return NRF24L01P_NG_ARD_250US;

    238 return (nrf24l01p_ng_ard_t)((retr_delay / 250) - 1);

    286static inline
    287int8_t nrf24l01p_ng_etoval_tx_power(nrf24l01p_ng_tx_power_t power)
    288{
    289 if (power == NRF24L01P_NG_TX_POWER_MINUS_18DBM) {
    290 return -18;
    291 }
    292 if (power == NRF24L01P_NG_TX_POWER_MINUS_12DBM) {
    293 return -12;
    294 }
    295 if (power == NRF24L01P_NG_TX_POWER_MINUS_6DBM) {
    296 return -6;
    297 }
    298 return 0;
    299}
