    647static inline void at86rf215_enable_radio(at86rf215_t *dev, uint8_t modulation)
    648{
    649 /* 16 bit frame-checksum, baseband enabled, checksum calculated by chip,
    650 frames with invalid cs are dropped */
    651 at86rf215_reg_write(dev, dev->BBC->RG_PC, modulation | PC_BBEN_MASK
    652 | PC_FCST_MASK | PC_TXAFCS_MASK
    653 | PC_FCSFE_MASK);
    654}
