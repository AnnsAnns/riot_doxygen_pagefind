    245static inline uint32_t gnrc_sixlowpan_frag_sfr_congure_snd_inter_frame_gap(
    246 gnrc_sixlowpan_frag_fb_t *fb
    247 )
    248{
    249#if IS_USED(MODULE_GNRC_SIXLOWPAN_FRAG_SFR_CONGURE)
    250 if (fb != NULL) {
    251 congure_snd_t *c = fb->sfr.congure;
    252 int32_t res = c->driver->inter_msg_interval(
    253 c, GNRC_SIXLOWPAN_FRAG_SFR_CONGURE_UNIT
    254 );
    255
    256 if ((res >= 0) &&
    257 ((unsigned)res >= CONFIG_GNRC_SIXLOWPAN_SFR_INTER_FRAME_GAP_US)) {
    258 return (uint32_t)res;
    259 }
    260 }
    261#else
    262 (void)fb;
    263#endif
