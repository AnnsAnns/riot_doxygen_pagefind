    549static inline void ina3221_config_set_enabled_channels(uint16_t *cfg, ina3221_ch_t ch)
    550{
    551 assert(cfg);
    552 uint16_t u16 = ((ch & INA3221_CH1) ? INA3221_ENABLE_CH1 : 0) |
    553 ((ch & INA3221_CH2) ? INA3221_ENABLE_CH2 : 0) |
    554 ((ch & INA3221_CH3) ? INA3221_ENABLE_CH3 : 0);
    555 *cfg &= ~INA3221_ENABLE_CH_MASK;
    556 *cfg |= (u16 & INA3221_ENABLE_CH_MASK);
